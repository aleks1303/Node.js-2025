const path = require('node:path')
const fsPromises = require('node:fs/promises');


const createFolders = async () => {

    const folder = 5
    const file = 5

    for (let i = 1; i <= folder; i++) {
       const folderPath = path.join(__dirname, 'baseFolder', `folder${i}`);
       await fsPromises.mkdir(folderPath,  {recursive: true});
        console.log(path.dirname(folderPath))
        console.log(__dirname, `folder${i}`)
        const stat = await fsPromises.stat(folderPath)
        console.log(stat.isFile())
        for (let j = 1; j <= file; j++) {
            const filePath = path.join(__dirname, 'baseFolder', `folder${i}`, `fileText_${j}.txt`)
            await fsPromises.writeFile(filePath, `Текст для ${i} папки та ${j} файлу`)
            console.log(__filename)
            const stat = await fsPromises.stat(filePath)
            console.log(stat.isFile())
        }
    }
}
